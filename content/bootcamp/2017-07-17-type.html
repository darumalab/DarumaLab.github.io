数値データ

num <- 7

オブジェクトが数値データなのかチェック

is.numeric(num)
## [1] TRUE
class(num)
## [1] "numeric"

オブジェクトが整数なのかチェック

is.integer(num)
## [1] FALSE
num2 <- 7L
is.integer(num2)
## [1] TRUE

文字列データ

char <- "テキスト"
char2 <- factor("テキスト")  #文字列をファクターとして保存

オブジェクトが文字列データなのかチェック

is.character(char)
## [1] TRUE
class(char)
## [1] "character"

論理データ

logi <- TRUE
logi2 <- FALSE
logi3 <- NA

オブジェクトが論理データなのかチェック

is.logical(logi)
## [1] TRUE
